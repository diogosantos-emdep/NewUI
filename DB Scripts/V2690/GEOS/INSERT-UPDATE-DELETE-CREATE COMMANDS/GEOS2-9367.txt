CREATE TABLE geos.`Transport_Frequency_Changelog` (
  `IdChangeLog` int unsigned NOT NULL AUTO_INCREMENT,
  `IdTransportFrequency` int NOT NULL,
  `Datetime` datetime NOT NULL,
  `IdUser` smallint unsigned NOT NULL,
  `Change` varchar(10000) DEFAULT NULL,
  PRIMARY KEY (`IdChangeLog`),  
  CONSTRAINT `transportfrequency_changelog_ibfk_2` FOREIGN KEY (`IdUser`) REFERENCES `users` (`IdUser`)
) ;
